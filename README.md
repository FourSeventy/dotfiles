## Setup
Symlink dotfiles

cd ~
ln -s ~/.bash_profile .dotfiles/.bash_profile
ln -s ~/.bashrc .dotfiles/.bashrc
ln -s ~/.curlrc .dotfiles/.curlrc
ln -s ~/.gitconfig .dotfiles/.gitconfig
